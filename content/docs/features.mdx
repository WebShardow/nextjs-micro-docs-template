# ทดสอบ Syntax Highlighting

นี่คือตัวอย่างโค้ด TypeScript:

```typescript
// นี่คือโค้ดตัวอย่าง
export async function fetchData(id: number): Promise<User> {
    const res = await fetch(`/api/users/${id}`);
    if (!res.ok) {
        throw new Error('Fetch failed');
    }
    return res.json();
}
